                        //Add new file to state list
                        files_state.push(STATE_IDLE);
                        files.push(file);
                        var currentindex = files_state.length - 1;

                        if (file.name.toLowerCase().indexOf(".apk") > -1) {
                            // Check md5 for this file
                            // Check for the various File API support.
                            if (window.File && window.FileReader && window.FileList && window.Blob) {
                                // Great success! All the File APIs are supported.
                                // Only check md5 if browser supports it
                                console.log("start md5");
                                files_state[currentindex] = STATE_CHECKING;
                                updateUI();

                                var hash = md5.create();
                                var chunk_size = 10 * 1024 * 1024; //10MB
                                var start = 0;
                                var stop = 0;

                                stop = chunk_size - 1;
                                if (stop > (file.size - 1)) {
                                    stop = (file.size - 1);
                                }

                                // Read file
                                var reader = new FileReader();

                                            start = stop + 1;

                                            stop = stop + chunk_size;
                                            if (stop > (file.size - 1)) {
                                                stop = (file.size - 1);
                                            }

                                            var blob = file.slice(start, stop + 1);
                                            reader.readAsArrayBuffer(blob);

                function updateUI(){
                    var idle_count = 0;
                    var checking_count = 0;
                    var checked_count = 0;
                    var uploading_count = 0;
                    var success_count = 0;
                    var fail_count = 0;

                    for(var i = 0; i < files_state.length; i++){
                        switch(files_state[i]){
                            case STATE_IDLE:
                                idle_count++;
                                break;
                            case STATE_CHECKING:
                                checking_count++;
                                break;
                            case STATE_CHECKED:
                                checked_count++;
                                break;
                            case STATE_UPLOADING:
                                uploading_count++;
                                break;
                            case STATE_SUCCESS:
                                success_count++;
                                break;
                            case STATE_FAIL:
                                fail_count++;
                                break;
                        }
                    }

                    var total_count = idle_count + checking_count + checked_count + uploading_count + success_count + fail_count;

                    //Start uploading again when files are added and an upload is in progress
                    if(idle_count + checking_count + checked_count > 0 && uploading_count > 0){
                        uploadrequested = true;
                    }

                    //Start upload as soon as files are ready to be uploaded and nothing is currently uploading.
                    if(idle_count + checking_count + uploading_count == 0 && uploadrequested){
                        uploadrequested = false;

                        if(checked_count > 0) {
                            myDropzone.processQueue();
                            return;
                        }
                    }

                    //Hide the upload cloud if there are any files in the queue
                    if(total_count > 0){
                        jQuery(".uploadCloud").hide();
                    }else{
                        jQuery(".uploadCloud").show();
                    }

                    //Show close only when queue is done (only successes and fails) and when there are no files at all
                    if((idle_count + checked_count + checking_count > 0 && uploading_count == 0) || ((idle_count + checked_count + checking_count + uploading_count == 0) && (success_count + fail_count > 0)) || total_count == 0){
                        //Show cancel only when queue is busy with something (uploading), or upload has been requested
                        if(uploading_count > 0 || uploadrequested){
                            jQuery("#btnDismiss").show().text("Cancel").removeClass("disabled");
                        }else{
                            jQuery("#btnDismiss").show().text("Close").removeClass("disabled");
                        }
                    }else{
                        //Show cancel only when queue is busy with something (uploading), or upload has been requested
                        if(uploading_count > 0 || uploadrequested){
                            jQuery("#btnDismiss").show().text("Cancel").removeClass("disabled");
                        }else{
                            jQuery("#btnDismiss").show().text("Close").addClass("disabled");
                        }
                    }




                    //Show upload only when files have been added and are not uploading yet and no upload has been requested
                    if(idle_count + checking_count + checked_count > 0 && !uploadrequested && uploading_count == 0){
                        jQuery("#btnSubmit").show().removeClass('disabled');
                    }else{
                        jQuery("#btnSubmit").show().addClass('disabled');

                    }

                    //Show success banner when there are files with success, and nothing is currently uploading or added
                    if(success_count > 0 && (idle_count + checking_count + checked_count + uploading_count == 0)){
                        jQuery("#success-message-bar").show("fast");
                    }else{
                        jQuery("#success-message-bar").hide("fast");
                    }

                    //Show error banner when there are files with errors
                    if(fail_count > 0 && (idle_count + checking_count + checked_count + uploading_count == 0)){
                        jQuery("#errorMessage").text("Some APKs had problems uploading.");
                        jQuery("#error-message-bar").show("fast");
                    }else{
                        jQuery("#error-message-bar").hide("fast");
                    }

                    //Show upload progress when there are files being uploaded
                    if(uploading_count > 0){
                        jQuery(".upload-progress").slideDown('fast');
                    }else{
                        jQuery(".upload-progress").slideUp('fast');
                    }
                }

                function uploadFilesSubmit() {
                    if(!jQuery("#btnSubmit").hasClass('disabled')) {

                        //Check if all fields have been filled out
                        var nameField = document.getElementById("fullname");
                        var emailField = document.getElementById("email");
                        var changesField = document.getElementById("changes");

                        jQuery("#error-message-bar").hide("fast");

                        if (nameField.value == "") {
                            jQuery("#errorMessage").text("Please enter your name.");
                            jQuery("#error-message-bar").show("fast");
                        }
                        else if (!validateEmail(emailField.value)) {
                            jQuery("#errorMessage").text("Please enter a valid email address.");
                            jQuery("#error-message-bar").show("fast");
                        }
                        else {
                            //Save data in cookies
                            setCookie("apkmirror_name", nameField.value, 1000);
                            setCookie("apkmirror_email", emailField.value, 1000);

                            uploadrequested = true;
                            updateUI();
                        }
                    }
                }

                function validateEmail(email) {
                    var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
                    return re.test(email);
                }

                function resetForm()
                {
                    if(!jQuery("#btnDismiss").hasClass("disabled")) {
                        myDropzone.removeAllFiles(true);

                        var changesField = document.getElementById("changes");
                        changesField.value = "";

                        files_state.length = 0;
                        files.length = 0;
                        updateUI();
                    }
                }



                function setCookie(cname, cvalue, exdays) {
                    var d = new Date();
                    d.setTime(d.getTime() + (exdays*24*60*60*1000));
                    var expires = "expires="+d.toUTCString();
                    document.cookie = cname + "=" + cvalue + "; " + expires + ";path=/";
                }

                function getCookie(cname) {
                    var name = cname + "=";
                    var ca = document.cookie.split(';');
                    for(var i=0; i<ca.length; i++) {
                        var c = ca[i];
                        while (c.charAt(0)==' ') c = c.substring(1);
                        if (c.indexOf(name) != -1) return c.substring(name.length,c.length);
                    }
                    return "";
                }
